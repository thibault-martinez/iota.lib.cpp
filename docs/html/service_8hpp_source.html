45 class Service {
46 public:
47  Service(const std::string& host, const uint16_t& port);
48  virtual ~Service();
49 
50 public:
51  template <typename Request, typename Response, typename... Args>
52  Response request(Args&&... args) const {
53  auto request = Request(args...);
54 
55  json data;
56  request.serialize(data);
57 
58  auto url = cpr::Url{ "http://" + host_ + ":" + std::to_string(port_) };
59  auto body = cpr::Body{ data.dump() };
60  auto headers = cpr::Header{ { "Content-Type", "application/json" },
61  { "Content-Length", std::to_string(body.size()) },
62  { "X-IOTA-API-Version", APIVersion } };
63  auto res = cpr::Post(url, body, headers);
64 
65  Response response;
66  // TODO set duration ?
67  if (res.status_code == 200)
68  response.deserialize(json::parse(res.text));
69  else if (res.status_code == 400)
70  throw Errors::BadRequest(json::parse(res.text)["error"].get<std::string>());
71  else if (res.status_code == 401)
72  throw Errors::Unauthorized(json::parse(res.text)["error"].get<std::string>());
73  else if (res.status_code == 500)
74  throw Errors::InternalServerError(json::parse(res.text)["error"].get<std::string>());
75  else
76  throw Errors::Unrecognized(json::parse(res.text)["error"].get<std::string>());
77 
78  return response;
79  }
80 
81 private:
82  std::string host_;
83  unsigned int port_;
84 };
