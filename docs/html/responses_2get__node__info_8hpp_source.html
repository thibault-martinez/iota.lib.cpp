53 public:
54  /*
55  * Name of the IOTA software you're currently using (IRI stands for
56  * Initial Reference Implementation).
57  */
58  const std::string& getAppName() const;
59  /*
60  * The version of the IOTA software you're currently running.

62  const std::string& getAppVersion() const;
63  /*
64  * Available cores on your machine for JRE.
65  */
66  const int64_t& getjreAvailableProcessors() const;

68  * Returns the amount of free memory in the Java Virtual Machine.
69  */
70  const int64_t& getJreFreeMemory() const;

72  * Returns the maximum amount of memory that the Java virtual machine will
73  * attempt to use.
74  */
75  const int64_t& getJreMaxMemory() const;
76  /*
77  * Returns the total amount of memory in the Java virtual machine.

81  * Latest milestone that was signed off by the coordinator.
82  */
83  const std::string& getLatestMilestone() const;
84  /*
85  * Index of the latest milestone.
86  */
87  const int64_t& getLatestMilestoneIndex() const;
88  /*
89  * The latest milestone which is solid and is used for sending transactions.
90  * For a milestone to become solid your local node must basically approve the
91  * subtangle of coordinator-approved transactions, and have a consistent view
92  * of all referenced transactions.
93  */
94  const std::string& getLatestSolidSubtangleMilestone() const;
95  /*
96  * Index of the latest solid subtangle.
97  */
98  const int64_t& getLatestSolidSubtangleMilestoneIndex() const;
99  /*
100  * Number of neighbors you are directly connected with.
101  */
102  const int64_t& getNeighbors() const;
103  /*
104  * Packets which are currently queued up.
105  */
106  const int64_t& getPacketsQueueSize() const;
107  /*
108  * Current UNIX timestamp.
109  */
110  const int64_t& getTime() const;
111  /*
112  * Number of tips in the network.
113  */
114  const int64_t& getTips() const;
115  /*
116  * Transactions to request during syncing process.
117  */
118  const int64_t& getTransactionsToRequest() const;
119 
120 private:
121  std::string appName_;

123  int64_t jreAvailableProcessors_;
124  int64_t jreFreeMemory_;

126  int64_t jreTotalMemory_;

135  int64_t transactionsToRequest_;
